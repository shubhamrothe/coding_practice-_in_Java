import static java.lang.System.out;
import java.util.*;
public class Myclass
{
    public static void main(String[] args)
    {
        Scanner sc=new Scanner(System.in);
        out.print("Enter the size of array");
        int size=sc.nextInt();
        int[] a;
        a=new int[size];
        int l=a.length;
        int sum=0;
        for(int i=0;i<l;i++)
        {
            a[i]=sc.nextInt();
            sum=sum+a[i];
        }
        out.println(sum);
        float Avg=(float)sum/l;
